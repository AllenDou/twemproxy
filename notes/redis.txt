## Examples

    append key1 value1
    '*3\r\n$6\r\nappend\r\n$4\r\nkey1\r\n$6\r\nvalue1'

    decr key1
    '*2\r\n$4\r\nDECR\r\n$4\r\nkey1\r\n'

    decrby key1 100
    '*3\r\n$6\r\nDECRBY\r\n$4\r\nkey1\r\n$3\r\n100\r\n'

    exits key1
    '*2\r\n$6\r\nEXISTS\r\n$4\r\nkey1\r\n'

    expire key1 1000
    '*3\r\n$6\r\nEXPIRE\r\n$4\r\nkey1\r\n$4\r\n1000\r\n'

    expireat key1 1327507603
    '*3\r\n$8\r\nEXPIREAT\r\n$4\r\nkey1\r\n$10\r\n1327507603\r\n'

    get key1
    '*2\r\n$3\r\nGET\r\n$4\r\nkey1\r\n'

    getbit key1 10
    '*3\r\n$6\r\nGETBIT\r\n$4\r\nkey1\r\n$2\r\n10\r\n'

    getrange key1 0 2
    '*4\r\n$8\r\nGETRANGE\r\n$4\r\nkey1\r\n$1\r\n1\r\n$1\r\n2\r\n'

    getset key1 myvalue1
    '*3\r\n$6\r\nGETSET\r\n$4\r\nkey1\r\n$8\r\nmyvalue1\r\n'

    hdel myhash field1field2 field3 field4
    '*6\r\n$4\r\nHDEL\r\n$6\r\nmyhash\r\n$6\r\nfield1\r\n$6\r\nfield2\r\n$6\r\nfield3\r\n$6\r\nfield4\r\n'

    hexists myhash field1
    '*3\r\n$7\r\nHEXISTS\r\n$6\r\nmyhash\r\n$6\r\nfield1\r\n'

    hget myhash field1
    '*3\r\n$4\r\nHGET\r\n$6\r\nmyhash\r\n$6\r\nfield1\r\n'

    hgetall myhash
    '*2\r\n$7\r\mHGETALL\r\n$6\r\nmyhash\r\n'

    hincrby myhash field1 1000
    '*4\r\n$7\r\nHINCRBY\r\n$6\r\nmyhash\r\n$6\r\nfield1\r\n$4\r\n1000\r\n' r\n'

    hkeys myhash
    '*2\r\n$5\r\nHKEYS\r\n$6\r\nmyhash\r\n'

    hlen myhash
    '*2\r\n$4\r\nHLEN\r\n$6\r\nmyhash\r\n'

    hmget myhash field1field2 field3 field4
    '*6\r\n$5\r\nHMGET\r\n$6\r\nmyhash\r\n$6\r\nfield1\r\n$6\r\nfield2\r\n$6\r\nfield3\r\n$6\r\nfield4\r\n'

    hmset myhash field1 value1 field2 value2 field3 value3 field4 value4
    '*10\r\n$5\r\nHMSET\r\n$6\r\nmyhash\r\n$6\r\nfield1\r\n$6\r\nvalue1\r\n$6\r\nfield2\r\n$6\r\nvalue2\r\n$6\r\nfield3\r\n$6\r\nvalue3\r\n$6\r\nfield4\r\n$6\r\nvalue4\r\n'

    hset myhash field1 value1
    '*4\r\n$4\r\nHSET\r\n$6\r\nmyhash\r\n$6\r\nfield1\r\n$6\r\nvalue1\r\n'

    hsetnx myhash field1 value1
    '*4\r\n$6\r\nHSETNX\r\n$6\r\nmyhash\r\n$6\r\nfield1\r\n$6\r\nvalue1\r\n'

    hvals myhash
    '*2\r\n$5\r\nHVALS\r\n$6\r\nmyhash\r\n'

    incr key1
    '*2\r\n$4\r\nINCR\r\n$4\r\nkey1\r\n'

    incrby key1 100
    '*3\r\n$6\r\nINCRBY\r\n$4\r\nkey1\r\n$3\r\n100\r\n'

    lindex mylist 2
    '*3\r\n$6\r\nLINDEX\r\n$6\r\nmylist\r\n$1\r\n2\r\n'

    linsert mylist BEFORE value1 value0
    '*5\r\n$7\r\nLINSERT\r\n$6\r\nmylist\r\n$6\r\nBEFORE\r\n$6\r\nvalue1\r\n$6\r\nvalue0\r\n'

    llen mylist
    '*2\r\n$4\r\nLLEN\r\n$6\r\nmylist\r\n'

    lpop mylist
    '*2\r\n$4\r\nLPOP\r\n$6\r\nmylist\r\n'

    lpush mylist value1 value2 value3
    '*5\r\n$5\r\nLPUSH\r\n$6\r\nmylist\r\n$6\r\nvalue1\r\n$6\r\nvalue2\r\n$6\r\nvalue3\r\n'

    lpushx mylist valuex
    '*3\r\n$6\r\nLPUSHX\r\n$6\r\nmylist\r\n$6\r\nvaluex\r\n'

    lrange mylist 0 2
    '*4\r\n$6\r\nLRANGE\r\n$6\r\nmylist\r\n$1\r\n0\r\n$1\r\n2\r\n'

    lrem mylist 1 valuex
    '*4\r\n$4\r\nLREM\r\n$6\r\nmylist\r\n$1\r\n1\r\n$6\r\nvaluex\r\n'

    lset mylist 0 value1
    '*4\r\n$4\r\nLSET\r\n$6\r\nmylist\r\n$1\r\n0\r\n$6\r\nvalue1\r\n'

    ltrim mylist 0 4
    '*4\r\n$5\r\nLTRIM\r\n$6\r\nmylist\r\n$1\r\n0\r\n$1\r\n4\r\n'

    move key1 db1
    '*3\r\n$4\r\nMOVE\r\n$4\r\nkey1\r\n$3\r\ndb1\r\n'

    persist key1
    '*2\r\n$7\r\nPERSIST\r\n$4\r\nkey1\r\n'

    rpop mylist
    '*2\r\n$4\r\nRPOP\r\n$6\r\nmylist\r\n'

    rpush mylist value1 value2 value3
    '*5\r\n$5\r\nRPUSH\r\n$6\r\nmylist\r\n$6\r\nvalue1\r\n$6\r\nvalue2\r\n$6\r\nvalue3\r\n'

    rpushx mylist valuex
    '*3\r\n$6\r\nRPUSHX\r\n$6\r\nmylist\r\n$6\r\nvaluex\r\n'

    sadd myset member1 member2 member3
    '*5\r\n$4\r\nSADD\r\n$5\r\nmyset\r\n$7\r\nmember1\r\n$7\r\nmember2\r\n$7\r\nmember3\r\n'

    scard myset
    '*2\r\n$5\r\nSCARD\r\n$5\r\nmyset\r\n'

    set key1 1000
    '*3\r\n$3\r\nSET\r\n$4\r\nkey1\r\n$4\r\n1000\r\n'

    setbit key1 1 1
    '*4\r\n$6\r\nSETBIT\r\n$4\r\nkey1\r\n$1\r\n1\r\n$1\r\n1\r\n'

    setex key1 100 1000
    '*4\r\n$5\r\nSETEX\r\n$4\r\nkey1\r\n$3\r\n100\r\n$4\r\n1000\r\n'

    setnx key1 1000
    '*3\r\n$5\r\nSETNX\r\n$4\r\nkey1\r\n$4\r\n1000\r\n'

    setrange key1 2 3
    '*4\r\n$8\r\nSETRANGE\r\n$4\r\nkey1\r\n$1\r\n2\r\n$1\r\n3\r\n'

    sismember myset member1
    '*3\r\n$9\r\nSISMEMBER\r\n$5\r\nmyset\r\n$7\r\nmember1\r\n'

    smembers myset
    '*2\r\n$8\r\nSMEMBERS\r\n$5\r\nmyset\r\n'

    spop myset
    '*2\r\n$4\r\nSPOP\r\n$5\r\nmyset\r\n'

    srandmember myset
    '*2\r\n$11\r\nSRANDMEMBER\r\n$5\r\nmyset\r\n'

    srem myset member1 memberx
    '*4\r\n$4\r\nSREM\r\n$5\r\nmyset\r\n$7\r\nmember1\r\n$7\r\nmemberx\r\n'

    strlen key1
    '*2\r\n$6\r\nSTRLEN\r\n$4\r\nkey1\r\n'

    ttl key1
    '*2\r\n$3\r\nTTL\r\n$4\r\nkey1\r\n'

    type key1
    '*2\r\n$4\r\nTYPE\r\n$4\r\nkey1\r\n'

    mget key1
    '*2\r\n$4\r\nMGET\r\n$4\r\nkey1\r\n'

    mget key1 key2
    '*3\r\n$4\r\nMGET\r\n$4\r\nkey1\r\n$4\r\nkey2\r\n'

    '*4\r\n$4\r\nMGET\r\n$4\r\nkey1\r\n$4\r\nkey2\r\n$4\r\nkey3\r\n'

    '*10\r\n$4\r\nMGET\r\n$4\r\nkey0\r\n$4\r\nkey1\r\n$4\r\nkey2\r\n$4\r\nkey4\r\n$4\r\nkey5\r\n$4\r\nkey6\r\n$4\r\nkey7\r\n $4\r\nkey8\r\n$4\r\nkey9\r\n'
